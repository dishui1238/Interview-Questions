var maxIceCream = function (costs, coins) {
  costs.sort((a, b) => a - b);

  let pay = 0;
  for (let i = 0; i < costs.length; i++) {
    const need = pay + costs[i];
    if (need > coins) {
      return i;
    }
    pay += costs[i];
  }
  return costs.length;
};
